Array Method

- length : 배열 길이를 구함
	var arr = new Array(3);
	alert(arr.length);
	OUTPUT : 3

- concat : 배열 병합
	var arr = new Array(0, 1, 2);
	var arr2 = new Array(4, 5, 6);
	arr.concat(arr2);
	OUTPUT : 012456

- join : 배열 원소들을 구분 기호로 합침
	var arr = new Array("일", "이", "삼");
	var arr2 = arr.join('+');
	OUTPUT : 일+이+삼

- shift : 배열의 첫번째 원소 삭제
	var arr = new Array(1, 2, 3, 4, 5);
	arr.shift();
	alert(arr);
OUTPUT : 2345

- unshift : 배열 처음에 원소를 추가
	var arr = new Array(1, 2, 3, 4, 5);
	arr.unshift(0);
	OUTPUT : 012345

- pop : 배열의 마지막 원소를 삭제
	var arr = new Array(1, 2, 3, 4, 5);
	arr.pop();
	OUTPUT : 1234

- push : 배열 끝에 원소를 추가
	var arr = new Array(1, 2, 3);
	arr.push(4, 5, 6, 7);
	OUTPUT : 1234567

- slice : 배열의 일정 범위를 추출
	var arr = new Array(1, 2, 3, 4, 5, 6, 7);
	var arr2 = arr.slice(3, 5);
	OUTPUT : 456

- sort : 배열을 문자순으로 정렬
	var arr = new Array(1, 2, 3, 13, 23);		// 3 >'1'3
	var arr2 = arr.sort();
	OUTPUT : 1132233

- reverse : 배열을 역순으로 정렬
	var arr = new Array(1, 2, 3, 13, 23);
	var arr2 = arr.reverse();
	OUTPUT : 3232131

- split : 문자열을 배열로 변환
	var str = "안.녕.하.세.요";
	var arr = str.split('.');
	OUTPUT : 안녕하세요
